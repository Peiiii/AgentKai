import { Config, Memory, Decision, Goal, GoalStatus } from '../types';
import { OpenAIModel } from '../models/OpenAIModel';
import { MemorySystem } from '../memory/MemorySystem';
import { DecisionEngine } from '../decision/DecisionEngine';
import { GoalManager } from '../goals/GoalManager';
import { v4 as uuidv4 } from 'uuid';
 
interface SystemResponse extends Decision {
    relevantMemories: Memory[];
    activeGoals: Goal[];
}
 
export class AISystem {
    private model: OpenAIModel;
    private memory: MemorySystem;
    private decision: DecisionEngine;
    private goals: GoalManager;
    private config: Config;
 
    constructor(config: Config) {
        this.config = config;
        this.model = new OpenAIModel(config.modelConfig);
        this.memory = new MemorySystem(this.model, config.memoryConfig);
        this.decision = new DecisionEngine(this.model, config.decisionConfig);
        this.goals = new GoalManager();
    }
 
    async processInput(input: string): Promise<SystemResponse> {
        // 获取相关记忆
        const relevantMemories = await this.memory.searchMemories(input);
 
        // 获取当前活跃目标
        const activeGoals = await this.goals.getActiveGoals();
 
        // 生成系统提示
        const systemPrompt = this.buildSystemPrompt(relevantMemories, activeGoals);
 
        // 做出决策
        const decision = await this.decision.makeDecision(input, systemPrompt);
 
        // 存储新记忆
        await this.memory.addMemory({
            id: uuidv4(),
            text: input,
            timestamp: Date.now(),
            metadata: {
                decision: decision.id,
                confidence: decision.confidence
            }
        });
 
        return {
            ...decision,
            relevantMemories,
            activeGoals
        };
    }
 
    async addGoal(params: Omit<Goal, 'id' | 'createdAt' | 'status' | 'progress' | 'updatedAt' | 'completedAt'>): Promise<Goal> {
        return this.goals.addGoal(params);
    }
 
    async getGoal(id: string): Promise<Goal | null> {
        return this.goals.getGoal(id);
    }
 
    async updateGoalStatus(id: string, status: GoalStatus): Promise<void> {
        await this.goals.updateGoal(id, { status });
    }
 
    async searchMemories(query: string): Promise<Memory[]> {
        return this.memory.searchMemories(query);
    }
 
    async getAllMemories(): Promise<Memory[]> {
        return this.memory.getAllMemories();
    }
 
    async getActiveGoals(): Promise<Goal[]> {
        return this.goals.getActiveGoals();
    }
 
    private buildSystemPrompt(memories: Memory[], goals: Goal[]): string {
        return `
系统状态:
- 记忆数量: ${memories.length}
- 活跃目标数量: ${goals.length}
 
当前目标:
${goals.map(g => `- ${g.description} (优先级: ${g.priority})`).join('\n')}
 
相关记忆:
${memories.map(m => `- ${m.text}`).join('\n')}
 
请基于以上信息做出合适的决策。
        `.trim();
    }
}
